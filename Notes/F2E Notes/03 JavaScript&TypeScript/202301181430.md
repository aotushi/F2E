### 实现Set方法

```javascript
//https://juejin.cn/post/7033275515880341512#:~:text=%F0%9F%9B%AB%20ES6%E7%AF%87-,%E5%AE%9E%E7%8E%B0set,-class%20Set%20%7B%0A%20%20constructor

class Set {
  constructor() {
    this.items = {};
    this.size = 0;
  }
  has(ele) {
    return ele in this.items;
  }
  
  add(ele) {
    if (!this.has(ele)) {
      this.items[ele] = ele;
      this.size++;
    }
    return this;
  }
  delete(ele) {
    if (this.has(ele)) {
      delete this.items[ele];
      this.size--;
    }
    return this;
  }
  
  clear() {
    this.items = [];
    this.size = 0;
  }
  
  values() {
    let values = [];
    for (let key in this.items) {
      if (this.items.hasOwnProperty(key)) {
        values.push(key);
      }
    }
    return values;
  }
}
```
